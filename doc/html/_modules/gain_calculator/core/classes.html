[docs]class LevelTerm:
    """
    Simple comparable data class representing a term of form 2p+(1)1
    where the last number is 2*J (J = total angular momentum over all shells).
    The rest of the shell notation is documented elsewhere.

    :ivar shell: shell object instance
    :ivar J2: double the value of total angular momentum
    """
    def __init__(self, shell, J2):  # type: (Shell, int) -> None
        """
        Init using a shell and total angular momentum
        :param shell: shell object instance
        :param J2: double the value of total angular momentum
        """
        self.shell = shell
        self.J2 = J2

    def __repr__(self):  # type: () -> str
        return str(self.shell) + str(self.J2)

    def __eq__(self, other):
        return str(self) == str(other)

    def __ne__(self, other):  # type: (LevelTerm, LevelTerm) -> bool
        return not self == other




[docs]class Shell:
    """
    Class representing a single shell of form 3d+4(2), where 3 is the principal quantum number
    d is the orbital quantum number (2) + represents spin direction (up),
    4 is electron count and 2 is double the value of angular momentum of the shell.

    :ivar principal_quantum_number: Starting from 1
    :ivar orbital_quantum_number: Given by value, use Shell.orbital_numbers for convenience
    :ivar spin_direction: expects "+" or "-" representing spin up and down
    :ivar double_angular_momentum: 2*J value (of shell only) e.g. two electrons up give 2*J=2
    :ivar electron_count: number of electrons in the shell

    :cvar orbital_numbers: mapping of orbital letters to numbers
    :cvar orbital_letters: inverse mapping to orbital numbers
    :cvar spin_direction_sign: mapping from "-" to -1 and "+" to 1
    """
    def __init__(self, principal_quantum_number, orbital_quantum_number, spin_direction,
                 double_angular_momentum, electron_count):  # type: (int, int, str, int, int) -> None
        """
        Initialize it based on quantum numbers spin and electron count. Consider using
        a factory method create from string for convenience.
        :param principal_quantum_number: Starting from 1
        :param orbital_quantum_number: Given by value, use Shell.orbital_numbers for convenience
        :param spin_direction: expects "+" or "-" representing spin up and down
        :param double_angular_momentum: 2*J value (of shell only) e.g. two electrons up give 2*J=2
        :param electron_count: number of electrons in the shell
        """
        self.principal_quantum_number = principal_quantum_number
        self.orbital_quantum_number = orbital_quantum_number
        self.spin_direction = spin_direction
        self.double_angular_momentum = double_angular_momentum
        self.electron_count = electron_count

    def __repr__(self):  # type: () -> str
        return str(self.principal_quantum_number) + \
               self.orbital_letters[self.orbital_quantum_number] + \
               self.spin_direction + \
               str(self.electron_count) + \
               "(" + str(self.double_angular_momentum) + ")"

    def __lt__(self, other):
        return self.get_num_value() < other.get_num_value()

    def __le__(self, other):
        return self.get_num_value() <= other.get_num_value()

    def __gt__(self, other):
        return not self <= other

    def __ge__(self, other):
        return not self < other

    def __eq__(self, other):  # type: (Shell, Shell) -> bool
        return str(self) == str(other)

    def __ne__(self, other):  # type: (Shell, Shell) -> bool
        return not self == other

[docs]    def get_num_value(self):
        """
        Assigns shell(shallow - refer to shallow_compare) a numerical value to allow shallow comparision
        :return: 100*n + 10*l + spin_sign (-1, 1), eg. 2p+ gives 211, 3d- gives 319
        """
        n = self.principal_quantum_number
        l = self.orbital_quantum_number
        spin_sign = self.spin_direction_sign[self.spin_direction]
        return 100 * n + 10 * l + spin_sign



[docs]    def shallow_compare(self, other):  # type: (Shell, Shell) -> bool
        """
        Check if the gives shells are shallow equal meaning 2p+3(1) == 2p+2(0) is true under shallow comparision
        :param other: shell to compare with
        :return: true if they shallow equal
        """
        return str(self)[:3] == str(other)[:3]

    def create_from_string(shell_repr):  # type: (str) -> Shell
        """
        Factory method to deconstruct a string of form 3d+4(2) and create a Shell instance based on it
        :param shell_repr: string to deconstruct
        :return: instance of Shell
        """
        return Shell(
            principal_quantum_number=int(shell_repr[0]),
            orbital_quantum_number=Shell.orbital_numbers[shell_repr[1]],
            spin_direction=shell_repr[2],
            double_angular_momentum=int(shell_repr[5]),
            electron_count=int(shell_repr[3])
        )


    orbital_numbers = {"s": 0, "p": 1, "d": 2, "f": 3, "g": 4, "h": 5, "i": 6}
    orbital_letters = {number: letter for letter, number in orbital_numbers.iteritems()}
    spin_direction_sign = {"+": 1, "-": -1}




[docs]class EnergyLevel:
    """
        Comparable class representing a single energy level.

        :ivar configuration: list of terms representing the full energy level configuration
        :ivar total_angular_momentum: number representing 2*all the angular momentum added
        :ivar relevant_terms: list of LevelTerm instances representing the whole configuration
    """
    def __init__(self, relevant_terms):  # type: (list) -> None
        """
        Init using list of instances of LevelTerm representing terms that are not closed
        :param relevant_terms: list of LevelTerm instances
        """
        self.relevant_terms = relevant_terms  #: terms giving the whole configuration
        self.total_angular_momentum, self.configuration = self.__generate_configuration()

    def __repr__(self):  # type: () -> str
        return ".".join(map(lambda term: str(term.shell) + str(term.J2), self.relevant_terms))

    def __eq__(self, other):  # type: (EnergyLevel, EnergyLevel) -> bool
        return str(self) == str(other)

    def __ne__(self, other):  # type: (EnergyLevel, EnergyLevel) -> bool
        return not self == other


[docs]    @staticmethod
    def create_from_string(energy_level_repr):  # type: (str) -> EnergyLevel
        """
        Factory method deconstructing a string 2s+1(1)1.3d+3(3)4 into terms then generating closed configuration
        up to last term and then replacing the proper terms in this configuration with relevant terms
        :param energy_level_repr: string to deconstruct
        :return: EnergyLevel instance
        """
        return EnergyLevel(relevant_terms=map(lambda term_string: LevelTerm(
            shell=Shell.create_from_string(term_string[:-1]),
            J2=term_string[-1]
        ), energy_level_repr.split(".")))


    @staticmethod
    def __get_shell_electron_count(orbital_quantum_number, spin_direction):  # type: (int, str) -> int
        j2 = 2 * orbital_quantum_number + Shell.spin_direction_sign[spin_direction]
        return j2 + 1

    @staticmethod
    def __create_full_shell(principal_quantum_number, orbital_quantum_number, spin_direction):
        electron_count = EnergyLevel.__get_shell_electron_count(orbital_quantum_number, spin_direction)
        if electron_count <= 0:
            return None
        return Shell(
            principal_quantum_number=principal_quantum_number,
            orbital_quantum_number=orbital_quantum_number,
            spin_direction=spin_direction,
            double_angular_momentum=0,
            electron_count=electron_count
        )

    @staticmethod
    def __get_orbital_numbers(principal_quantum_number):  # type: (int) -> list
        return range(-principal_quantum_number + 1, principal_quantum_number)

    @staticmethod
    def __generate_shells(principal_quantum_number):
        n = principal_quantum_number
        el = EnergyLevel
        return [el.__create_full_shell(n, l, s_d) for s_d in ["-", "+"] for l in el.__get_orbital_numbers(n)]

    @staticmethod
    def __generate_closed_configuration(last_shell):  # type: (Shell) -> list
        n = last_shell.principal_quantum_number
        el = EnergyLevel
        shells = itertools.chain.from_iterable([el.__generate_shells(i) for i in range(1, n + 1)])
        shells = sorted(filter(lambda shell: shell is not None, shells))
        return filter(lambda shell: shell <= last_shell, shells)

    @staticmethod
    def __is_relevant(shell, relevant_shell):
        return relevant_shell.shallow_compare(shell)

    def __replace_relevant(self, shell):
        relevant_terms = [term for term in self.relevant_terms if self.__is_relevant(shell, term.shell)]
        if len(relevant_terms) == 0:
            return shell
        elif len(relevant_terms) == 1:
            return relevant_terms[0].shell
        else:
            raise Exception("Duplicate relevant terms!")

    def __remap_total_angular_momentum(self, shell):  # type: (Shell) -> LevelTerm
        # TODO look into this, might be a mistake only adding angular momentum, is subtraction possible?
        self.total_angular_momentum = self.total_angular_momentum + shell.double_angular_momentum
        return LevelTerm(shell, J2=self.total_angular_momentum)

    def __generate_configuration(self):  # type: () -> tuple
        shells = EnergyLevel.__generate_closed_configuration(self.relevant_terms[-1].shell)
        shells = map(self.__replace_relevant, shells)
        self.total_angular_momentum = 0
        return self.total_angular_momentum, map(self.__remap_total_angular_momentum, shells)
